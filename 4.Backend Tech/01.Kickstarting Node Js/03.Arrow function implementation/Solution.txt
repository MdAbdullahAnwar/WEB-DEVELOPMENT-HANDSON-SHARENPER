-Attribute: 
     -count
 -Method:
     -increment 
     -incrementWithDelay

const Counter = {
    // your implementation here
    count: 0,

    increment: function () {
        this.count++;
        return this.count;
    },

    incrementWithDelay: function(delay){
        setTimeout(() => {
            this.count++;
            console.log(this.count);
        },delay)
    }
};


console.log(Counter.increment());    
console.log(Counter.increment());    
Counter.incrementWithDelay(2000);
