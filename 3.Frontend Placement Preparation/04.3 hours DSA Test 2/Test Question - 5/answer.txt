#include<iostream>
#include<vector>
#include<queue>
#include<cmath>
using namespace std;

static int result[1000];

int* sharpees(int n)
{
/* Function to store first n sharpees in an array
Return the array containing the sharpees*/
   
    int count = 0;
    queue<int> q;

    for(int i=1;i<=9; i++){
        q.push(i);
    }

    while(!q.empty() && count < n){
        int num = q.front();
        q.pop();

        if(num >= 10)  {
            result[count++] = num;
        }

        int last_digit = num % 10;

        if(last_digit > 0){
            q.push(num * 10+(last_digit - 1));
        }

        if(last_digit < 9){
            q.push(num * 10 + (last_digit + 1));
        }
    }
    return result;    
/* Dont change anything below. If changed click on reset. */

}

int main()
{
    int n;
    cin>>n;
    int * out =sharpees(n);
    for(int i=0;i<n;i++)
    {
        cout<<out[i]<<endl;
    }
    
    return 0;
}
