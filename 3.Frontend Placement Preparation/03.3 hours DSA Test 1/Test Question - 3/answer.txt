#include <iostream>
#include <sstream>
#include <vector>
#include <algorithm>

using namespace std;
class Solution {
public:
    string reverse(string s) {
        vector<string> words;
        stringstream ss(s);
        string word;

        while(ss >> word){
            words.push_back(word);
        }

        std::reverse(words.begin(), words.end());

        string result;

        for(int i=0; i<words.size(); ++i){
            result += words[i];
            if(i != words.size()-1){
                result += " ";
            }
        }
        return result;
    }
};
